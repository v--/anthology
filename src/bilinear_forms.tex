\subsection{Bilinear forms}\label{subsec:bilinear_forms}

\begin{definition}\label{def:bilinear_form}\cite[249]{Knapp2016BAlg}
  Let \( V \) be an \( F \)-vector space. A \Def{bilinear form} over \( V \) is a bilinear\Tinyref{def:multilinear_function} functional\Tinyref{def:linear_operator}, that is, a multilinear function of type \( V \times V \to F \).
\end{definition}
