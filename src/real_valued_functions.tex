\subsection{Real-valued functions}\label{subsec:real_valued_functions}

\begin{definition}\label{def:epigraph}
  Let \( X \) be an arbitrary set. The \Def{epigraph} of the function \( f: X \to \BR \) is defined as
  \begin{equation*}
    \Epi f \coloneqq \{ (x, r) \in X \times \BR \colon r \geq f(x) \},
  \end{equation*}
\end{definition}

\begin{definition}\label{def:function_definiteness}
  Let \( f: X \to \BR \) be a real-valued functional from a real or complex vector space \( X \).

  \begin{itemize}
    \item \( f \) is \Def{positive definite} if \( f(x) > 0 \) for all \( x \in X \setminus \{ 0_X \} \) and \( f(0_X) = 0 \).
    \item \( f \) is \Def{negative definite} if \( f(x) < 0 \) for all \( x \in X \setminus \{ 0_X \} \) and \( f(0_X) = 0 \).
    \item \( f \) is \Def{positive semidefinite} if \( f(x) \geq 0 \) for all \( x \in X \setminus \{ 0_X \} \) and \( f(0_X) = 0 \).
    \item \( f \) is \Def{negative semidefinite} if \( f(x) \leq 0 \) for all \( x \in X \setminus \{ 0_X \} \) and \( f(0_X) = 0 \).
  \end{itemize}
\end{definition}
