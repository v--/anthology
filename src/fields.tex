\subsection{Fields}\label{subsec:fields}

\begin{definition}\label{def:field}
  As mentioned in \cref{def:semiring/field}, fields are commutative division rings.
\end{definition}

\begin{definition}\label{def:field_extension}
  If \( F \) and \( G \) are fields and \( F \) is a unital subring\Tinyref{def:first_order_structure/substructure} of \( G \), we say that \( F \) is a \Def{subfield} of \( G \) and that \( G \) is a \Def{field extension} of \( F \). If \( G = F \), we say that \( G \) is a \Def{trivial field extension} of \( F \).

  Field extension are also denoted as \( G / F \) to highlight the roles of \( G \) and \( F \). This is not a quotient ring but simply a notation. See \cref{def:galois_group}.
\end{definition}

\begin{definition}\label{def:galois_group}\cite[124]{Knapp2016BAlg}
  Let \( F \) be a field extension\Tinyref{def:field_extension} of \( G \). The group \( \Gal(F / G) \) of automorphisms of \( F \) that leave \( G \) fixed is called the \Def{Galois group} of the field extension \( F / G \).
\end{definition}

\begin{definition}\label{def:field_characteristic}
  We define the \Def{characteristic} of a field \( F \) as
  \begin{itemize}
    \item the smallest number of times \( 1 \) must be added to itself in order to obtain \( 0 \).
    \item zero if \( 0 \) cannot be obtained in this way.
  \end{itemize}
\end{definition}

\begin{theorem}\label{thm:ring_of_integers_module_prime_is_field}
  The ring \( \Z_n \)\Tinyref{def:ring_of_integers_modulo} of integers modulo \( n \) is a field if \( n \) is a prime number\Tinyref{def:prime_number}.
\end{theorem}
\begin{proof}
  We only need to show that \( \Z_n \) has a multiplicative inverse for any nonzero element.

  Fix \( x \in \Z_n \). If \( y \) is a multiplicative inverse of \( x \), we should have
  \begin{equation*}
    xy \equiv 1 \pmod n,
  \end{equation*}
  which is the same as
  \begin{equation*}
    n \mid (xy - 1).
  \end{equation*}

  \Cref{thm:bezout_identity} gives us integers \( a, b \in \Z \) such that
  \begin{equation*}
    ax + bn = \gcd(x, n) = 1,
  \end{equation*}
  which is the same as
  \begin{equation*}
    -bn = xa - 1.
  \end{equation*}

  Define \( y \coloneqq \Rem(a, n) \). This is the multiplicative inverse of \( x \).
\end{proof}

\begin{definition}\label{def:transcendetal_element}\cite[454]{Knapp2016BAlg}
  We say that the element \( a \in G \) of the field extension \( G \) of \( F \) is \Def{transcendental} over \( F \) if either of the equivalent conditions hold:
  \begin{defenum}
    \DItem{def:transcendetal_element/evaluation} The evaluation map\Tinyref{thm:polynomial_ring_universal_property} \( \Phi_a: F[X] \to F[a] \) is injective.

    \DItem{def:transcendetal_element/polynomial} There exists no polynomial \( p(X) \in F[X] \) such that \( p(a) = 0 \).
  \end{defenum}

  If \( a \) is not transcendental, we say that is is \Def{algebraic}.
\end{definition}

\begin{definition}\label{def:algebraic_extension}\cite[456]{Knapp2016BAlg}
  We say that the field extension \( G \) of \( F \) is an \Def{algebraic extension} if every element of \( G \) is algebraic over \( F \).
\end{definition}

\begin{proposition}\label{thm:field_elements_are_algebraic}
  Every field is an algebraic\Tinyref{def:algebraic_extension} of itself.
\end{proposition}
\begin{proof}
  If \( a \in F \), then \( F[a] = F \) because every polynomial evaluates to some real number, depending on \( a \), and the constant polynomials already take all possible values. Thus \( \Phi_a \) is not injective and all elements from a field are algebraic over the field itself.
\end{proof}

\begin{theorem}\label{thm:algebraic_extension_always_exists}\cite[485]{Knapp2016BAlg}
  If \( p(X) \) is a prime polynomial over the field \( F \), there exists an algebraic extension of \( F \) of the form
  \begin{equation*}
    G = F[u],
  \end{equation*}
  where \( u \in G \) is a root of \( p(X) \).
\end{theorem}
\begin{proof}
  Since \( p(X) \) is prime, \( \Gen{p(X)} \) is a nontrivial prime ideal and by \cref{def:prime_ring_ideal}, the quotient \( G \coloneqq F / \Gen{p(X)} \) is a field. It is an extension field of \( F \).

  Define
  \begin{equation*}
    u \coloneqq X + \Gen{p(X)}.
  \end{equation*}

  Then
  \begin{equation*}
    p(u) = p(X + \Gen{p(X)}) = p(X) + \Gen{p(X)} = \Gen{p(X)},
  \end{equation*}
  hence \( u \) is a root of \( p(X) \) in \( G \).

  Thus \( u \) is algebraic over \( F \) and \( G \) is an algebraic extension of \( F \).

  It remains to show that \( G = F[u] \). First, take a coset \( q(X) + \Gen{p(X)} \) in \( G \). We have
  \begin{equation*}
    q(X) + \Gen{p(X)} = (q(X) - X) + (X + \Gen{p(X)}) = (q(X) - X) + u,
  \end{equation*}
  hence this belongs in \( F[u] \). Conversely, evaluating a polynomial \( q(X) \in F[X] \) at \( u \) gives us
  \begin{equation*}
    q(u) = q(X + \Gen{p(X)}) = q(X) + \Gen{p(X)},
  \end{equation*}
  which is a coset of \( G \).
\end{proof}

\begin{theorem}\label{thm:e_is_transcendental}\label{thm:eulers_constant_is_transcendental}
  Euler's constant\Tinyref{def:eulers_constant} \( e \) is transcendental over \( \Q \).
\end{theorem}

\begin{theorem}\label{thm:pi_is_transcendental}\cite[454]{Knapp2016BAlg}
  The number \( \pi \)\Tinyref{def:pi} is transcendental over \( \Q \).
\end{theorem}

\begin{example}\label{ex:polynomials_over_pi}
  \Cref{thm:pi_is_transcendental} implies that the polynomials \( \Q[X] \) can be embedded into \( \R \) via \( \Phi_\pi: \Q[X] \to \R \). We can identify a polynomial
  \begin{equation*}
    p(X) = \sum_{i=0}^n a_i X^i
  \end{equation*}
  with rational coefficients with
  \begin{equation*}
    p(\pi) = \sum_{i=0}^n a_i \pi^i.
  \end{equation*}
\end{example}

\begin{definition}\label{def:splitting_field}\cite[458]{Knapp2016BAlg}
  We say that a polynomial \( p(X) \in F[X] \) over the field \( F \) \Def{splits} if \( p(X) \) can be factored\Tinyref{def:factorization_in_ring} into a product of linear polynomials in \( F[X] \).

  A \Def{splitting field} of \( p(X) \) over \( F \) is a field extension \( G \) such that
  \begin{itemize}
    \item \( p(X) \) splits over \( G \).
    \item \( G \) is generated\Tinyref{def:generated_ring_ideal} by \( F \) and the roots of \( p(X) \) over \( G \).
  \end{itemize}
\end{definition}

\begin{definition}\label{def:algebraically_closed_field}\cite[proposition 9.20]{Knapp2016BAlg}
  We say that the field \( F \) is algebraically closed if any of the equivalent conditions are satisfied:
  \begin{defenum}
    \DItem{def:algebraically_closed_field/trivial_algebraic_extensions} \( F \) has no nontrivial algebraic extensions\Tinyref{def:algebraic_extension}.
    \DItem{def:algebraically_closed_field/linear_irreducible_polynomials} Every irreducible polynomial in \( F[X] \) is linear.
    \DItem{def:algebraically_closed_field/at_least_one_root} Every nonconstant polynomial in \( F[X] \) has at least one root in \( F \).
    \DItem{def:algebraically_closed_field/factorization} Every polynomial in \( F[X] \) factors\Tinyref{def:factorization_in_ring} into a product linear polynomials.
    \DItem{def:algebraically_closed_field/exactly_n_roots} Every polynomial in \( F[X] \) of degree \( n \) has exactly \( n \) roots in \( F \).
  \end{defenum}
\end{definition}
\begin{proof}
  \begin{description}
    \Implies[def:algebraically_closed_field/trivial_algebraic_extension][def:algebraically_closed_field/linear_irreducible_polynomials] Let \( p(X) \) be an irreducible polynomial in \( F[X] \). By \cref{thm:pid_prime_iff_irreducible}, \( p(X) \) is prime. By \cref{thm:algebraic_extension_always_exists}, there exists an algebraic extension \( G \) of \( F \) such that the prime polynomial \( p(X) \) has a root in \( G \). But \( F \) has no nontrivial algebraic extensions, hence \( F = G \) and \( p(X) \) has a root \( u \in F \).

    If \( p(X) \) is not linear\LEM, we can divide \( p(X) \) by \( (X - u) \) to obtain a lower-degree non-constant polynomial. Hence \( p(X) \) is linear.

    \Implies[def:algebraically_closed_field/linear_irreducible_polynomials][def:algebraically_closed_field/factorization] With induction on the polynomial degree, we split a polynomial \( p(X) \) into a product of linear polynomials.

    This is obvious for \( \deg p = 1 \). Assume that the statement holds for polynomial of degree strictly less than \( n \) and let \( p(X) \) be a polynomial of degree \( n \). By \cref{def:factorization_in_ring}, it is reducible\Tinyref{def:irreducible_ring_element}, that is, there exist non-invertible (that is, non-constant) polynomials \( r_1(X) \) and \( r_2(X) \), such that
    \begin{equation*}
      p(X) = r_1(X) r_2(X).
    \end{equation*}

    Since both \( r_1(X) \) and \( r_2(X) \) are non-constant, they have a positive degree less than \( n \). Hence the induction hypothesis holds for them and both can be factored into linear polynomials. Therefore their product \( p(X) \) can also be factored into linear polynomials.

    This completes the proof.

    \Implies[def:algebraically_closed_field/at_least_one_root][def:algebraically_closed_field/factorization] Suppose that \( u_1 \) is a root of \( p(X) \). \Cref{thm:polynomial_root_iff_divisible} tells us that \( p(X) \) is divisible by \( (X - u_1) \). Using induction on the degree of \( p(X) \), we can factor \( p(X) \) into
    \begin{equation*}
      p(X) = a (X - u_1) (X - u_2) \cdots (X - u_n),
    \end{equation*}
    where \( a \in F \).

    \Implies[def:algebraically_closed_field/factorization][def:algebraically_closed_field/exactly_n_roots] Follows from \cref{thm:polynomial_root_iff_divisible} Follows from \cref{thm:polynomial_root_iff_divisible} by induction on the polynomial degree. Note that the number of roots is bounded by \( n \) (see \cref{thm:integral_domain_polynomial_root_limit}).

    \Implies[def:algebraically_closed_field/exactly_n_roots][def:algebraically_closed_field/trivial_algebraic_extensions] By \cref{thm:integral_domain_polynomial_root_limit}, if \( p(X) \) has degree \( n \) and exactly \( n \) roots, then it has no more roots. Hence all roots of \( p(X) \) are already in the field \( F \) and \( F \) is the only algebraic extension of itself.
  \end{description}
\end{proof}
